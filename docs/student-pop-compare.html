from collections import OrderedDict

student_metrics = pd.read_csv('./graduated_students.csv',index_col=0)

# fix up the GPAs greater than 5: just assign to median for now
student_metrics[student_metrics['HSGPA'] > 5] = student_metrics['HSGPA'].median()

all_stats = student_metrics.describe()
deans_list = student_metrics[student_metrics.cumgpa >= 3.5].describe()
below35 = student_metrics[student_metrics.cumgpa < 3.5].describe()
ontime = student_metrics[student_metrics.time2grad <= 4].describe()
late = student_metrics[student_metrics.time2grad > 4].describe()

groups = OrderedDict()
groups['all'] = all_stats
groups['deans_list'] = deans_list
groups['below_3.5'] = below35
groups['4year-'] = ontime
groups['4year+'] = late

# build data frame; making standardized test proportions manageable
metric_means = np.array([g.loc['mean'].values for g in groups.values()])
means = pd.DataFrame(metric_means.T, columns=groups.keys(), index=student_metrics.columns)

toplot = means.copy()
toplot = toplot.drop('SAT_Total_1600').drop('TOEFL_Convert_Paper').drop('ACT_Composite')

metric_error = np.array([g.loc['std'].values for g in groups.values()])
errors = pd.DataFrame(metric_error.T, columns=groups.keys(), index=student_metrics.columns)
# toplot.loc['SAT_Total_1600'] = toplot.loc['SAT_Total_1600'] / 1600 * 10
# toplot.loc['TOEFL_Convert_Paper'] = toplot.loc['TOEFL_Convert_Paper'] / 677 * 10
# toplot.loc['ACT_Composite'] = toplot.loc['ACT_Composite'] / 32 * 10

# bar plot data for comparison across subpopulations
fig, ax = plt.subplots(figsize=(16,10))
toplot.plot(ax=ax, kind='bar', yerr=metric_error[::,:6])
ax.xaxis.grid(False)
ax.set_title("Comparison of Graduated Students Across Population Groupings")
ax.set_ylabel("Values")
fig.savefig("graduated-students-compare.pdf", bbox_inches="tight", dpi=300)

def standardize(df, col):
    """Standardize a particular column in the data frame."""
    df.loc[col] = ((df.loc[col] - df.loc[col].mean()) /
                   (df.loc[col].max() - df.loc[col].min()))

# build data frame; standardizing all student features
toplot = means.copy()
for label in toplot.index:
    standardize(toplot, label)
    
# bar plot data for comparison across subpopulations
fig, ax = plt.subplots(figsize=(16,10))
toplot.plot(ax=ax, kind='bar')
ax.xaxis.grid(False)
ax.set_title("Standardized Comparison of Graduated Students Across Population Groupings")
ax.set_ylabel("Values")
fig.savefig("graduated-students-compare-standardized.pdf", bbox_inches="tight", dpi=300)
